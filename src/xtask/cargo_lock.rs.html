3use std::process::Command;
4
5use crate::DynError;
6
7pub(crate) fn cargo_lock() -> Result<(), DynError> {
8    println!("Checking Cargo lock");
9    let status = Command::new("cargo")
10        .current_dir(crate::project_root())
11        .args(["tree", "--locked"])
12        .env("RUSTFLAGS", "-Cpanic=abort -Zpanic_abort_tests")
13        .stdout(std::process::Stdio::null())
14        .status()?;
15
16    if !status.success() {
17        return Err("cargo tree --locked failed; Please include required changes to Cargo.lock in your pull request".into());
18    }
19    Ok(())
20}
